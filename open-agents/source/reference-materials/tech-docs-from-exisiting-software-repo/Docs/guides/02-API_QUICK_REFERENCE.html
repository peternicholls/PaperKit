üöÄ Swift: Basics

Single Anchor with Preset Style

let journey = ColorJourney(
    config: .singleAnchor(
        ColorJourneyRGB(red: 0.5, green: 0.2, blue: 0.8),
        style: .balanced  // or .pastelDrift, .vividLoop, etc.
    )
)

Multi-Anchor Journey

let config = ColorJourneyConfig(
    anchors: [color1, color2, color3],
    loopMode: .closed
)
let journey = ColorJourney(config: config)

Sample Colors

// Single sample at position
let color = journey.sample(at: 0.5)  // t ‚àà [0, 1]

// Discrete palette
let colors = journey.discrete(count: 8)

// Gradient stops
let gradient = journey.linearGradient(stops: 20)

üîß C: Basics

Initialize and Create

#include "ColorJourney.h"

CJ_RGB anchor = {0.5f, 0.2f, 0.8f};
CJ_Config config = cj_config_init();
config.anchors[0] = anchor;
config.anchor_count = 1;

CJ_Journey *journey = cj_journey_create(&config);

Sample Colors

// Single sample
CJ_RGB color = cj_journey_sample(journey, 0.5f);

// Discrete palette
CJ_RGB *palette = malloc(8 * sizeof(CJ_RGB));
cj_journey_discrete(journey, 8, palette);

// Access
for (int i = 0; i < 8; i++) {
    CJ_RGB color = palette[i];
    printf("%.2f, %.2f, %.2f\n", color.r, color.g, color.b);
}

Cleanup

cj_journey_destroy(journey);
free(palette);

üìã Configuration Enums

LightnessBias

.neutral     // No change
.lighter     // Shift toward brighter
.darker      // Shift toward darker
.custom(weight: Float)  // [-1.0, 1.0]

ChromaBias (Saturation)

.neutral     // No change
.muted       // 0.6x saturation
.vivid       // 1.4x saturation
.custom(multiplier: Float)  // [0.5, 2.0]

Contrast

.low         // ŒîE ‚â• 0.05
.medium      // ŒîE ‚â• 0.1
.high        // ŒîE ‚â• 0.15
.custom(threshold: Float)  // Custom ŒîE

Temperature

.neutral     // No bias
.warm        // Reds, oranges, yellows
.cool        // Blues, greens, purples

LoopMode

.open        // Start and end distinct
.closed      // Loops back to start
.pingPong    // Reverses at boundaries

Variation

.off         // No variation
.subtle(dimensions: [.hue, .lightness], seed: UInt64)
.noticeable(dimensions: [.hue, .chroma], seed: UInt64)

üí° Common Patterns

Theme-Aware Palettes

// Light mode
let lightConfig = .singleAnchor(color, style: .balanced)
    .lightness(.lighter)

// Dark mode  
let darkConfig = .singleAnchor(color, style: .balanced)
    .lightness(.darker)

UI Element Categories

let journey = ColorJourney(
    config: .singleAnchor(baseColor, style: .balanced)
        .contrast(.high)
        .chroma(.vivid)
)
let colors = journey.discrete(count: categoryCount)

Smooth Gradients

let journey = ColorJourney(
    config: .singleAnchor(baseColor, style: .balanced)
)
let gradient = journey.linearGradient(stops: 50)

Data Visualization (Cool to Warm)

let config = ColorJourneyConfig(
    anchors: [blueColor, redColor],  // Cool to warm
    loopMode: .open
)
let colorMap = ColorJourney(config: config)

let lowColor = colorMap.sample(at: 0.0)   // Cool
let midColor = colorMap.sample(at: 0.5)   // Neutral
let highColor = colorMap.sample(at: 1.0)  // Warm

Seeded Variation for Consistency

let config = .singleAnchor(color, style: .balanced)
    .variation(.subtle(
        dimensions: [.hue, .lightness],
        seed: 42  // Same seed = same variation
    ))

// Always produces same palette
let palette1 = ColorJourney(config: config).discrete(count: 10)
let palette2 = ColorJourney(config: config).discrete(count: 10)
// palette1 == palette2 ‚úì
