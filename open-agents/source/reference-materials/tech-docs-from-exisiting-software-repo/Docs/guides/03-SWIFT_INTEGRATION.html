üìç Imports & Setup

Basic Import

import ColorJourney

With SwiftUI

import SwiftUI

Color Type Bridging

ColorJourney provides automatic conversion to platform color types:

let cgColor = ColorJourneyRGB(red: 0.5, green: 0.2, blue: 0.8)

// SwiftUI Color
let swiftUIColor: Color = cgColor.color

// UIKit UIColor (iOS)
let uiColor: UIColor = cgColor.uiColor

// AppKit NSColor (macOS)
let nsColor: NSColor = cgColor.nsColor

üí° Basic Usage

Create a Single-Anchor Journey

let journey = ColorJourney(
    config: .singleAnchor(
        ColorJourneyRGB(red: 0.5, green: 0.2, blue: 0.8),
        style: .balanced
    )
)

Sample a Single Color

// Sample at position (0.0 = start, 1.0 = end)
let midColor = journey.sample(at: 0.5)
let startColor = journey.sample(at: 0.0)
let endColor = journey.sample(at: 1.0)

Generate a Palette

// 8 distinct, perceptually uniform colors
let palette = journey.discrete(count: 8)

Create a Gradient

// SwiftUI-compatible gradient with 50 stops
let gradient = journey.linearGradient(stops: 50)

‚öôÔ∏è Configuration

Builder Pattern (Recommended)

let config = .singleAnchor(baseColor, style: .balanced)
    .lightness(.lighter)
    .chroma(.vivid)
    .contrast(.high)
    .temperature(.warm)

Full Configuration Object

let config = ColorJourneyConfig(
    anchors: [color1, color2, color3],
    lightness: .lighter,
    chroma: .vivid,
    contrast: .high,
    midJourneyVibrancy: 0.3,
    temperature: .warm,
    loopMode: .closed,
    variation: .subtle(
        dimensions: [.hue, .lightness],
        seed: 42
    )
)

Configuration Options

Lightness Bias

.neutral     // Keep original lightness
.lighter     // Shift toward brighter colors
.darker      // Shift toward darker colors
.custom(weight: 0.5)  // Custom [-1.0, 1.0]

Chroma Bias (Saturation)

.neutral     // Keep original saturation
.muted       // 60% saturation (0.6x)
.vivid       // 140% saturation (1.4x)
.custom(multiplier: 1.2)  // Custom [0.5, 2.0]

Contrast

.low         // Soft (ŒîE ‚â• 0.05)
.medium      // Balanced (ŒîE ‚â• 0.1)
.high        // Strong (ŒîE ‚â• 0.15)
.custom(threshold: 0.08)  // Custom ŒîE value

Temperature

.neutral     // No temperature bias
.warm        // Shift toward reds, oranges, yellows
.cool        // Shift toward blues, greens, purples

Loop Modes

.open        // Colors at t=0 and t=1 are different
.closed      // Color at t=1 matches t=0 (seamless loop)
.pingPong    // Reverses at boundaries

Variation

.off         // No variation

.subtle(
    dimensions: [.hue, .lightness],
    seed: 42
)

.noticeable(
    dimensions: [.hue, .chroma, .lightness],
    seed: 42
)

üé® SwiftUI Integration

Discrete Color Grid

struct ColorPaletteGrid: View {
    let journey = ColorJourney(
        config: .singleAnchor(baseColor, style: .balanced)
    )
    
    var body: some View {
        VStack(spacing: 0) {
            ForEach(journey.discrete(count: 8).indices, id: \.self) { i in
                Rectangle()
                    .fill(journey.discrete(count: 8)[i].color)
                    .frame(height: 50)
            }
        }
    }
}

Linear Gradient

struct GradientView: View {
    let journey = ColorJourney(
        config: .singleAnchor(baseColor, style: .balanced)
    )
    
    var body: some View {
        Rectangle()
            .fill(journey.linearGradient(stops: 20))
            .frame(height: 200)
    }
}

Dynamic Configuration

struct ThemedPalette: View {
    @Environment(\.colorScheme) var colorScheme
    
    var body: some View {
        let lightness: LightnessBias = colorScheme == .dark ? .darker : .lighter
        
        let config = .singleAnchor(baseColor, style: .balanced)
            .lightness(lightness)
        
        let journey = ColorJourney(config: config)
        
        VStack {
            ForEach(journey.discrete(count: 6).indices, id: \.self) { i in
                Rectangle()
                    .fill(journey.discrete(count: 6)[i].color)
            }
        }
    }
}

‚ö° Performance Tips

Create Once, Use Many Times

// ‚úì Good: Create outside view
let journey = ColorJourney(config: config)

struct MyView: View {
    var body: some View {
        let colors = journey.discrete(count: 10)
        // Use colors...
    }
}

// ‚úó Bad: Creating inside view body
struct MyView: View {
    var body: some View {
        let journey = ColorJourney(config: config)  // Recreates every frame!
        // Use journey...
    }
}

Cache Discrete Palettes

struct CachedPalette {
    let journey: ColorJourney
    let colors: [ColorJourneyRGB]
    
    init(config: ColorJourneyConfig, count: Int) {
        self.journey = ColorJourney(config: config)
        self.colors = journey.discrete(count: count)
    }
}

Sampling vs Discrete

// For smooth gradients: sample continuously
let gradient = journey.linearGradient(stops: 50)

// For UI elements: use discrete
let palette = journey.discrete(count: 8)

// For single colors: use sample
let color = journey.sample(at: 0.5)

üìö Code Examples

Material-Inspired Color System

struct MaterialColors {
    static let primary = ColorJourneyRGB(red: 0.2, green: 0.6, blue: 1.0)
    
    static func palette(style: JourneyStyle) -> [ColorJourneyRGB] {
        let journey = ColorJourney(
            config: .singleAnchor(primary, style: style)
                .contrast(.high)
        )
        return journey.discrete(count: 10)
    }
}

Theme-Aware Color Provider

class ColorProvider {
    @Environment(\.colorScheme) var colorScheme
    
    func palette(for anchor: ColorJourneyRGB) -> [ColorJourneyRGB] {
        let lightness: LightnessBias = colorScheme == .dark ? .darker : .lighter
        
        let config = ColorJourneyConfig(
            anchors: [anchor],
            lightness: lightness,
            chroma: .vivid,
            contrast: .high
        )
        
        return ColorJourney(config: config).discrete(count: 8)
    }
}

Chart Color Mapper

struct ChartColors {
    static let coolToWarm = ColorJourney(
        config: ColorJourneyConfig(
            anchors: [
                ColorJourneyRGB(red: 0.0, green: 0.0, blue: 1.0),  // Blue
                ColorJourneyRGB(red: 1.0, green: 0.0, blue: 0.0)   // Red
            ],
            loopMode: .open
        )
    )
    
    static func colorForValue(_ value: Float, min: Float, max: Float) -> Color {
        let normalized = (value - min) / (max - min)
        let clipped = max(0, min(1, normalized))
        return coolToWarm.sample(at: clipped).color
    }
}
