üé® Your First Colors

Swift

import ColorJourney

let journey = ColorJourney(
    config: .singleAnchor(
        ColorJourneyRGB(red: 0.5, green: 0.2, blue: 0.8),
        style: .balanced
    )
)

// Generate 8 distinct colors
let colors = journey.discrete(count: 8)

// Use in SwiftUI
ForEach(colors.indices, id: \.self) { i in
    Rectangle()
        .fill(colors[i].color)
        .frame(height: 50)
}

C

#include "ColorJourney.h"

CJ_RGB anchor = {0.5f, 0.2f, 0.8f};
CJ_Config config = cj_config_init();
config.anchors[0] = anchor;
config.anchor_count = 1;

CJ_Journey *journey = cj_journey_create(&config);

// Generate 8 colors
for (int i = 0; i < 8; i++) {
    float t = (float)i / 7.0f;
    CJ_RGB color = cj_journey_sample(journey, t);
    printf("%.2f, %.2f, %.2f\n", color.r, color.g, color.b);
}

cj_journey_destroy(journey);
‚úÖ What Happened?

ColorJourney created a smooth path through color space, generating 8 visually distinct, perceptually uniform colors from your single anchor color.

‚öôÔ∏è Configuration Basics

Adjust Brightness

// Dark mode
.singleAnchor(color, style: .balanced)
    .lightness(.darker)

// Light mode
.singleAnchor(color, style: .balanced)
    .lightness(.lighter)

Control Saturation

// Soft, pastel colors
.singleAnchor(color, style: .balanced)
    .chroma(.muted)

// Bold, vibrant colors
.singleAnchor(color, style: .balanced)
    .chroma(.vivid)

Enforce Contrast

// Soft, subtle
.singleAnchor(color, style: .balanced)
    .contrast(.low)

// High contrast, easy to read
.singleAnchor(color, style: .balanced)
    .contrast(.high)

Temperature Shift

// Warm and inviting
.singleAnchor(color, style: .balanced)
    .temperature(.warm)

// Cool and professional
.singleAnchor(color, style: .balanced)
    .temperature(.cool)

üí° Common Use Cases

Timeline Tracks

let journey = ColorJourney(
    config: .singleAnchor(baseColor, style: .balanced)
        .contrast(.high)
)
let trackColors = journey.discrete(count: 12)

Category Labels

let journey = ColorJourney(
    config: .singleAnchor(baseColor, style: .vividLoop)
        .chroma(.vivid)
)
let categoryColors = journey.discrete(count: 8)

Smooth Gradients

let journey = ColorJourney(
    config: .singleAnchor(baseColor, style: .balanced)
)
let gradient = journey.linearGradient(stops: 50)

Multi-Color Transitions

let config = ColorJourneyConfig(
    anchors: [redColor, greenColor, blueColor],
    loopMode: .closed
)
let palette = ColorJourney(config: config).discrete(count: 12)

üñ•Ô∏è SwiftUI Integration

import SwiftUI
import ColorJourney

struct ColorPaletteView: View {
    let journey = ColorJourney(
        config: .singleAnchor(
            ColorJourneyRGB(red: 0.4, green: 0.5, blue: 0.9),
            style: .balanced
        )
    )
    
    var body: some View {
        VStack(spacing: 0) {
            ForEach(journey.discrete(count: 8).indices, id: \.self) { index in
                Rectangle()
                    .fill(journey.discrete(count: 8)[index].color)
                    .frame(height: 40)
            }
        }
    }
}

‚ö†Ô∏è Common Mistakes to Avoid

‚ùå Wrong: Creating journeys repeatedly 
// Inefficient
for i in 0..<100 {
    let journey = ColorJourney(config: config)
    let color = journey.sample(at: Float(i) / 100)
}
‚úÖ Right: Create once, sample many times 
// Efficient
let journey = ColorJourney(config: config)
for i in 0..<100 {
    let color = journey.sample(at: Float(i) / 100)
}
‚ùå Wrong: Assuming RGB color math

RGB (0.5, 0.5, 0.5) is NOT 50% brightness! ColorJourney handles perceptual math internally.

‚úÖ Right: Trust ColorJourney's OKLab calculations

Just provide RGB colors and trust the output for perceptual correctness.

üìñ Key Concepts

Journey
A parametrized path through color space that defines how colors transition
Anchor
The starting color(s) that define the journey's direction
OKLab
A perceptually uniform color space where distances equal perceived color differences
Perceptually Uniform
Colors equally spaced in math are equally spaced visually to the human eye
ŒîE (Delta E)
Perceptual distance between two colors - ensures distinctness
Chroma
Saturation/colorfulness - controls vibrancy
Lightness
Brightness - controls how dark or light colors appear
